  83.     /**

  96.         $lastfield = false;
  97.         $fields    = explode("\n", preg_replace('/\x0D\x0A[\x09\x20]+/', ' ', $str));
  98.         foreach ($fields as $field) {
  99.             if (strlen(trim($field)) == 0) {
  100.                 $lastfield = ':body';
  101.             } elseif ($lastfield == ':body') {
  102.                 $retval['body'] .= $field."\n";
  103.             } else {
  104.                 if (($field[0] == ' ' or $field[0] == "\t") and $lastfield !== false) {
  105.                     $retval['headers'][$lastfield] .= ' '.$field;
  106.                 } elseif (preg_match('/([^:]+): (.+)/m', $field, $match)) {
  107.                     $match[1] = strtolower(preg_replace('/(?<=^|[\x09\x20\x2D])./e', 'strtoupper("\0")', strtolower(trim($match[1]))));

  111.                         } else {

  113.                         }
  114.                     } else {
