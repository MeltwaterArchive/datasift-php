  82.     /**
  83.      * Parse an HTTP response. Separates the headers from the body and puts
  84.      * the headers into an associative array.

  86.      * @param string $str The HTTP response to be parsed.

  88.      */
  89.     static private function parseHTTPResponse($str)
  90.     {
  91.         $retval = array(
  92.             'headers' => array(),

  94.         );
  95.         $lastfield = false;
  96.         $fields    = explode("\n", preg_replace('/\x0D\x0A[\x09\x20]+/', ' ', $str));
  97.         foreach ($fields as $field) {
  98.             if (strlen(trim($field)) == 0) {
  99.                 $lastfield = ':body';
  100.             } elseif ($lastfield == ':body') {
  101.                 $retval['body'] .= $field."\n";
  102.             } else {
  103.                 if (($field[0] == ' ' or $field[0] == "\t") and $lastfield !== false) {
  104.                     $retval['headers'][$lastfield] .= ' '.$field;
  105.                 } elseif (preg_match('/([^:]+): (.+)/m', $field, $match)) {
  106.                     $match[1] = strtolower(preg_replace('/(?<=^|[\x09\x20\x2D])./e', 'strtoupper("\0")', strtolower(trim($match[1]))));
  107.                     if (isset($retval['headers'][$match[1]])) {
  108.                         if (is_array($retval['headers'][$match[1]])) {
  109.                             $retval['headers'][$match[1]][] = $match[2];
  110.                         } else {
  111.                             $retval['headers'][$match[1]] = array($retval['headers'][$match[1]], $match[2]);
  112.                         }
  113.                     } else {
  114.                         $retval['headers'][$match[1]] = trim($match[2]);
  115.                     }
  116.                 }
  117.             }
  118.         }
