  56. function datasift_autoloader($classname)

  59.     static $libdirlen = false;
  60.     if ($libdir === false) {

  62.         $libdirlen = strlen($libdir);
  63.     }
  64.  
  65.     $retval = false;
  66.  
  67.     if (substr($classname, 0, 9) == 'DataSift_') {
  68.         // The realpath function will return false if the file does not exist
  69.         $filename = realpath($libdir.str_replace('_', '/', $classname).'.php');
  70.         if ($filename !== false) {
  71.             assert($libdir == substr($filename, 0, $libdirlen));
  72.             include $filename;
  73.             $retval = true;
  74.         }
  75.     }
