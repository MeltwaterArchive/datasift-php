  91.     {
  92.         do {
  93.             // Disconnect and reconnect
  94.             $this->reconnect();
  95.  
  96.             // Params for stream_select
  97.             $read   = array($this->_conn);
  98.             $write  = null;
  99.             $except = null;
  100.  
  101.             $buf = '';
  102.  
  103.             // Loop while the connection is valid using stream_select with a
  104.             // timeout. The timeout allows us to check to see if a stop has been
  105.             // requested.
  106.             while ($this->_conn && !feof($this->_conn) && ($count = stream_select($read, $write, $except, $this->_read_timeout))) {
  107.                 // Only continue if we're in the right state
  108.                 if ($this->_state != parent::STATE_RUNNING) {
  109.                     break;
  110.                 }
  111.  
  112.                 if (!$count) {
  113.                     // Nothing waiting, listen again
  114.                     continue;
  115.                 }
  116.  
  117.                 // Set the stream as blocking
  118.                 stream_set_blocking($this->_conn, 1);
  119.  
  120.                 // Get the chunk length
  121.                 $len = fgets($this->_conn, $this->_max_line_length);
  122.                 if ($len === false) {
  123.                     // EOF
  124.                     continue;
  125.                 }
  126.  
  127.                 // Read and process each line
  128.                 while ($this->_state == parent::STATE_RUNNING and ($line = fgets($this->_conn, $this->_max_line_length)) !== false) {
  129.                     if (strlen($line) == 0) {
  130.                         // End of the chunk
  131.                         break;
  132.                     }
  133.  
  134.                     // Decode the JSON
  135.                     $interaction = json_decode(trim($line), true);
  136.

  140.                         if (!empty($interaction['interaction'])) {
  141.                             $this->onInteraction($interaction);
  142.                         }
  143.                     }
  144.                 }
  145.  
  146.                 // Set the stream as non-blocking
  147.                 stream_set_blocking($this->_conn, 0);
  148.             }
  149.         } while ($this->_conn && !feof($this->_conn) and $this->_auto_reconnect and $this->_state == parent::STATE_RUNNING);

  236.                     $connection_delay++;

  239.                 }

  243.  
  244.                 if ($code == '200') {
  245.                     // Success!
  246.                     $this->_state = parent::STATE_RUNNING;
  247.                 } elseif ($code == '404') {
  248.                     // The hash doesn't exist
  249.                     throw new DataSift_Exception_StreamError('Hash not found!');
  250.                 } else {
  251.                     // Connection failed, back off a bit and try again
  252.                     // Timings from http://dev.datasift.com/docs/streaming-api
  253.                     if ($connection_delay == 0) {
  254.                         $connection_delay = 10;
  255.                     } elseif ($connection_delay < 320) {
  256.                         $connection_delay *= 2;
  257.                     } else {
  258.                         throw new DataSift_Exception_StreamError('Connection failed: '.$code.' '.$message);
  259.                     }
  260.                 }
  261.             }
  262.         } while ($this->_state != parent::STATE_RUNNING);
  263.  
  264.         // Set the stream as non-blocking
  265.         stream_set_blocking($this->_conn, 0);
  266.     }
